cmake_minimum_required(VERSION 3.5)
project(dronestack)

# Find dependencies
find_package(ament_cmake REQUIRED)
find_package(rclcpp REQUIRED)
find_package(std_msgs REQUIRED)
find_package(gazebo_msgs REQUIRED)
find_package(mavros_msgs REQUIRED)
find_package(Eigen3 REQUIRED)
find_package(geometry_msgs REQUIRED)
find_package(nav_msgs REQUIRED)
find_package(tf2 REQUIRED)
find_package(tf2_ros REQUIRED)
find_package(tf2_msgs REQUIRED)

# Specify C++17 standard
add_compile_options(-std=c++17)

# Include directories
include_directories(
  include
  ${EIGEN3_INCLUDE_DIR}
)

# Add executables
add_executable(offb_node_sitl src/offb_node_sitl.cpp)
ament_target_dependencies(offb_node_sitl rclcpp std_msgs gazebo_msgs geometry_msgs nav_msgs mavros_msgs tf2 tf2_ros tf2_msgs Eigen3)

# Install targets
install(TARGETS offb_node_sitl
  DESTINATION lib/${PROJECT_NAME})

# Install headers
install(DIRECTORY include/dronestack/
  DESTINATION include/${PROJECT_NAME})

ament_package()
